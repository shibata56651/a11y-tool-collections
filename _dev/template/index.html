JavaScriptサンプル

function initCarousel() { const carousel = document.querySelector('[data-carousel]'); if (!carousel) return; const slides = carousel.querySelectorAll('.mv__slide'); let currentSlide = 0; function showSlide(index) { slides.forEach((slide, i) => { slide.classList.toggle('mv__slide--active', i === index); }); } function nextSlide() { currentSlide = (currentSlide + 1) % slides.length; showSlide(currentSlide); } setInterval(nextSlide, 3000); }
